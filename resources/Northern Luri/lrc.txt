لٛۏر نوماْ یٱکؽ د لٛونؽا دۏنیا ٱ کاْ هان د ٱفتاونشیݩ ۉ هارگٱ ٱفتاونشیݩ ڤلات اؽروݩ ۉ ٱفتاوزنوݩ ۉ هارگٱ ٱفتاوزنوݩ ڤلات عراق ۉ شومال اومان. اؽ لٛو ڤ زڤون لٛۏری قسٱ مؽکٱن. فتراقیٛا دماترشناسی ۉ تۏرؽا پؽا بیٱ د آیمؽا دما ڤیرگار د ڤیرگار چلٛ هزار سالٱ د ڤلاتؽا لٛۏرنشیݩ دؽاری مؽکٱ. ڤا ڤیرداری ڤ بٱردنۏشتٱیا ۉ شؽڤٱیا ڤ دٱس اومایٱ د بارٱ خٱلق زیتری ایلٛام دؽاری مۊٱ کاْ ایلٛامیا « لٛۏریا دمایی» یٱنؽ، جٱد خٱلق لٛۏرسوݩ ایسنی د ٱفتاونشیݩ ڤلات اؽرونٱن. تۏرؽیا ڤ جا مٱنٱ د چوغازٱمبیلٛ، چوغامؽش ۉ هٱف چۊغا دؽاری مؽکٱ کاْ ایلٛامیا لٛۏریایٱن کاْ د شیش هزار سال دماتر د اؽ راساگٱ بینٱ. ٱڤازاریایی کاْ پؽا بینٱ دیاری مؽکٱن کاْ سی سالؽا فرٱ درازؽ ڤنو پاتشا ۉ هیزداری کۏیا لٛۏرسوݩ ۉ خۊزسوݩ نٱ. زڤون لٛۏری ڤا اوماین ماد یا آلشت بیٱ د زڤون اؽرونی. د خٱلق راساگٱیا لٛۏرنشیݩ مۊٱ بۉیی ایلٛامیان (3000 تا 559 ن.ڤ) کاْ ڤنو د راساگٱیا لٛۏرسوݩ، پوشتکۏ (ايلٛام)، خۊزسوݩ، بٱختؽوری، كوگیلۊیٱ ۉ بۊیرٱمٱد ۉ قٱرؽ د آستۊن فارس ایسنی زئشت مؽکردنٱ گیریشمٱن، گوتٱ پاتشایی ایلٛاميا د ڤٱر گرتٱ ڤلاتچٱیا ماسباتيك يا ماسباتيس (راساگٱ یا پوشتكۏ لٛۏرسوݩ يا ایلٛام ایسنی) سيماش (ٱفتاوزنوݩ اْسفاهوݩ ۉ گولپايگان ایسنی) شۊش (خۊزسوݩ ایسنی) آنشان (قٱری د خۊزسوݩ، اْسفاهوݩ ۉ فارس ایسنی) ۉ كربيانٱ (آستوݩ لٛۏرسوناْ ایسنی) دیاری کردٱ. کاسی یا کاْ ڤا ایلٛامیا قۉم بینٱ د راساگٱیا هارگٱ تر د لٛۏرسوݩ کۏیا شومالی تر جولگٱ شۊش د لٛۏرسوݩ ۉ د نزیکیٛا آستوݩ کرماشوݩ د مٱرز ٱفتاوزنوݩ سۊبی گالایان (د نزیکیا هارگٱ کرماشوݩ) زئشت داشتنٱ. د مؽنجایا هزارٱ ساْئم نۊئا ڤاڤن د سال ۲۴۰۰ ن. ڤ. د شومال ڤلات ایلٛام ۉ هارگٱ راساگٱ یا کاسی یا (د شومال خۊزسوݩ ۉ هارگٱ لٛۏرسوݩ) جاؽؽ هؽ کاْ نومش کاشن بیٱ کاْ ڤ نشونی د کاسیا نٱ د خوش دارٱ اؽ راساگٱ دماتر ڤ شٱئر لٛۏری دمادار د دۏرٱ یا اْسلامی آلشت بیٱ. کلٱمٱ ایسنی هۊز یا خۊز د کلٱمٱ هئسی یا اوکسی یٱنؽ نوم لٛون کاسی گرتٱ بیٱ. کلٱمٱیاؽؽ کاْ ایساْ ڤا نومؽا کۊشک، کۊشکی، کۊسٱ، کاسیو کاْ ڤ جایا ڤلات شناسی یا هۊزیا د کورسۊ مۉٱن هومبٱراڤٱرٱ ڤا لٛونؽیا نوم دؽار کاسیا. ایسنٱم یٱکؽ د هۊزؽا لٛۏرسوݩ رٱشنۊ ٱ کاْ نوم یٱکؽ د خودایا کاسی بیٱ. جنک شال ۉ سترٱ کاْ ایساْ د مؽن خٱلق لٛۏرسوݩ د ڤٱر مؽکٱن جنکؽٱ کاْ کاسیا د ڤٱر مؽکردنٱ.


سی ٱڤٱلی گاْلٛ نوم لٛۏر د سٱدٱ چوارم ڤ شیڤٱ اللریه، لاریه، اللور و لوریه اومائٱ. لٛۏر یاٛ نوم ڤاجٱ ایلٛامی-کاسیتٱ کاْ نوماْ شٱئر دمادار هئن گات کاسیت بیٱ کاْ لٛۏر ریشٱش ها د اؽ شٱئرآڤیشتاٛیی. نوم کورد نهاتر د اؽرشت ئاْرۉیا د اؽروݩ هومبراڤٱر بیٱ ڤا مال نشین ۉ ٱشایر ۉ هومبا لٛونؽ دیاری ناٛییٱ.[۱][۲][۳][۴][۵][۶][۷].[۸] د پایٱ ۉ یاٛ قٱرؽ د سٱرچٱشمٱیا، لٛۏریاۉ داٛیلٛٱمیاۉ بٱلۊچؽا ۉ خٱلقؽ هٱنی ۉ جگا د کوردیا هساو کردنٱ سی یٱ کاْ کلٱمؽیا"کورد" هومبراوڤٱر بیٱ ڤا جۊر زئشت نٱ هومبایی زڤون ۉ جٱرغٱ لٛۉنی.[۹] د دیجا لٛون لٛۏؽیا بونکۊیی جگا د کوردیا هساو مۊٱن.[۱۰]


هٱمدلا موستوفی مۉئٱ د زوبداٛ التٱڤاریخ اومائٱ: نوم لٛۏر سی یٱ ڤ اؽ لٛوݩ گوتٱ مۊٱ کاْ د ڤلات مارۊد دهاتؽٱ کاْ ڤش مۉٱن کورد ۉ دۉر ۉ گردش دٱبٱن ڤٱن ڤ زڤون لٛۏری کۊل ٱر بونٱن ۉ دش دٱربٱن جایی ٱ کاْ لٛۏر مونٱن سی یٱ کاْ ریشٱشو د او راساگٱ بیٱ ڤشو مۉٱن لٛۏر. کلٱمٱ دۏم ڤنٱ کاْ ڤ زڤون لٛۏری کۊ چر لئر مۉٱن ۉ سی سٱخت بیئن رٱ کٱسرٱ ناْ لام زٱمٱ کردن ۉ گوتنٱ لٛۏر ڤاجٱ ساْئاْم ینٱ کاْ اؽ لٛوݩ د نٱسلاْ کٱسؽٱن کاْ نومش لٛۏر بیٱ ۉ ڤٱ دۏرسترٱ.


خٱلق لٛۏر دۏ قٱلاْ فئرٱ نوم دیار ۉ فئرٱ گٱپ دارٱ کاْ مۊٱن ڤ لٛۏر گٱپ ۉ لٛۏر کوچک بٱر مۊٱن کاْ مؽنتلاشو گلال دز یا سزار دیاری مؽکٱ.


لٛۏریا بٱختیوری: کاْ خوشو دۏ تا دٱسٱ دارٱن چارلٛنڳ ۉ هٱفت لٛنڳ.بٱختیوریا هان د آستونؽا خۊزسوݩ ۉ آستون چارمال بٱختیوری ۉ آستونؽیا لٛۏرسوݩ، اْسفاهوݩ ۉ شٱئرؽیا پاؽمونی ۉ هارگٱ ٱفتاونشین اؽ آستون زناٛیی مؽکٱن.

لٛۏرؽیا پاؽمونی: د ڤٱرگرتٱ مٱمٱسٱنی٬ کوگیۊیٱ٬ لیراڤی ۉ جانٱکی ۉ کومزاری ٱ کاْ د ری میزونکاری د شٱمال ۉ ٱفتاونشین فارس٬ آستون کوگیلۊیٱ ۉ بویرٱمٱد٬ قاْرؽ د آستون خۊزسوݩ قٱل گٱپؽ د بۊشئر ۉ شٱمال ڤلات اومان زناٛیی مؽکٱن.

لٛۏرؽیا کوچک دۏ دٱسان فاٛیلٛی سٱلاسی.
لٛۏرؽیا فاٛیلٛی: د ڤٱرگرتٱ مؽنجایی٬ لٛٱک٬ مٱهٱکی یا مٱلکی ۉ کٱلهۏر ٱن. مؽنجایی یا هان د آستونؽا لٛۏرسوݩ، هٱمدوݩ، خۊزسوݩ، ایلٛام، مٱرکٱزی، قٱزڤین، ٱفتاوزنوݩ تؽروݩ، پاؽموناْ گیلٛان ۉ ٱفتاوزنوݩ ڤلات ئاْراق د آستونؽیا دیالٱ، ڤاست، ماٛیسان ۉ بٱغداد، لٛٱکؽیا هان د آسونؽیا لٛۏرسوݩ، ایلٛام (پوشتکۏ لٛۏرسوݩ)، کرماشوݩ؟ هٱمدوݩ ۉ کوردسوݩ، مٱهٱکیا هان د آستون ایلٛامۉ ٱفتاوزنوݩ ڤلات ئاْراق، کٱلهۏرؽیا هان د آستون کرماشوݩ، قٱرؽ د کوردسوݩ ۉ ٱفتاوزنوݩ ڤلات ئاْراق.

لٛۏرؽیا سٱلاسی: هان د آستونؽیا کرماشوݩ، هٱمدوݩ، لٛۏرسوݩ ۉ مٱرکٱزی.کوملۊس زڤونؽا لٛۏری یٱکؽ د زڤونؽا هؽن اورۊپایی ٱ ۉ د تٱلٛاْ اؽرونی ٱفتاونشینٱ کاْ ها د ڤلاتؽیا اؽروݩ،ئاْراق، اومان. قٱری د کلٱمؽیا لٛۏری ریشٱ دماداری دارٱن کاْ ریشٱشو مؽرٱسٱ ڤ زڤونؽا کاسی، ایلٛامی. زڤونؽا لٛۏری هان د ساْ پٱلٛ، لٛۏری فاٛیلٛی، لٛۏری بٱختیوری ۉ لٛۏری پاؽمونی(دومنی).


ٱڤٱلین خٱلقی کاْ د خاک لٛۏرسوݩ زئشت مؽکردنٱ ايلٛاميا بینٱ. ڤنو د ٱڤٱلیا هزارٱ چوارمی نئها ڤلات بیئن خوشو د راساگٱیا لٛۏرنشین رٱڤٱندیاری کردنٱ. دۉلٱت ايلٛام د ڤٱرگرتٱ خۊزسوݩ، لٛۏرسوݩ ایسنی، پوشتکۏ لٛۏرسوݩ ۉ کۏیا بٱختیوری بیٱ. خٱلق ايلٛام ڤ ڤلات خوشو آنزو یا آنسو یا سوسونکا مؽ گوتنٱ بابلیا د ڤلات آلام يا آلامتو هومبایی کۉگٱ ۉ گاسی ڤلات ٱفتاوزنوݩ مؽ گوتنٱ. شٱئرؽیا گٱپش ڤنونٱن: شۊش، ٱڤاز، ماداكتو (ساٛمرٱ – ]]) خايدالۊ كاْ د جا خورموئٱ ایسنی بیٱ. ايلٛامیا د گات چٱن هزار سال موتاج ٱسلی خوشوناْ ری ڤ ری لونؽا هیزداری چی سۊمر يا آكديا، بابليا ۉ آشۊر يا بیئن ۉ دٱس آخر سی جیاڤازیا مؽناْ خوشو ۉ جٱنڳیا هوناٛیی د دۏشمنی خوشو آشۊر تۊزٱنن (۶۵۴ ن.ڤ).


کاسی یا کاْ د کوتاْ شٱمال ڤلات ایلٛام زئشت مؽکردن ۉ د سوار کاری فرٱ دٱسلات داشتن ۵۷۶ سال د بابل پاتشایی کردن ۉ دٱس آخر دۉلٱت ایلٛام تۊزٱنشو ۉ د بابل کردشو ڤ دٱر. کاسی یا دماترؽ گلٛ هٱردن د بابل ۉ باٛنٱ نٱهراٛین تونسن نٱسٱقاْ مٱفرٱقاْ لٛۏرسوناْ راس کونی خوشوݩ نومدیار بٱکٱن جۊرؽ کاْرۊمٱن گریشمٱن دماترشناس نوم دیار مۊٱ کاْ نٱسٱق اؽروݩ دماتری یٱنؽ نٱسٱق مٱفرٱق کاری لٛۏرسوݩ. ٱلڤٱتٱ مۊٱ کاْ او سازیارؽا چئنی مٱفرٱقی کیمری یا بینٱ. کاسی یا ڤ پاتشایی یا دماتر د خوشو کارگرایی فرٱ اؽ نؽان چی رٱڤشت کوچک کردن شیڤٱیا، ڤنو نهارۉ نٱسٱق کاریا جونهٱرکٱش د گات هٱخامنشی بینٱ.


لٛۏرسوݩ د گات ساسانی ٱندوم ڤلات پٱهلٱ یا پٱهلو بی کاْ ئاْرۉیا دماتر ڤش مؽگوتنٱ جبال. ڤلات پٱهلٱ خوش هٱنی بٱر ماٛیی ڤ بٱرجایا کوچکترؽ. د شٱئرؽیا گٱپ کٱلون ۉ گات ساٛمرٱ ۉشاپۊرخاس بی کاْ ڤ دٱس شاپۊر یٱکوم رٱڤٱندیاری بی. لٛۏرسوݩ د سال ۲۱ مالکنوݩ دما گرتن ناڤٱن ڤ دٱس بوسٱلمونؽا شلارۊ بی. زلٛ بین ئاْرۉیا بوسٱلموݩ کاراگرایی زؽادؽ د رٱزیشت خٱلقاْ لٛۏرسوݩ ۉ راساگٱیا هٱنی نؽیا ڤا اؽ هال ۉ بال خٱلقاْ لٛۏر تونسن زڤون ۉ رٱزیشت خوشوناْ زنٱ ڤردارٱن. د سال ۳۰۰ مالکنوݩ لٛۏرسوݩ بی ڤ دۏ کوت لٛۏر گٱپ ۉ لٛۏر کوچک اؽ دۏ راساگٱ پاتشایی شو هوم گات ڤ دٱس دۏ برار بی، پاتشا لٛۏر گٱپ بٱدر ۉ پاتشا لٛۏر کوچک ٱبۊمٱنسۊر بی. مؽنجا ڤنو راساگٱیایی بی کاْ تا شیراز دیاری مؽکرد ڤش مؽگوتن شاولسو ۉ ایساْ ڤا نوم مٱمٱسٱنی نوم دیار بیٱ دۏ کوتش ڤ دٱس ٱتابٱکؽیا لٛۏر گٱپ ۉ ٱتابٱکؽیا لٛۏر کوچک دیڤۊنداری بیٱ بی. دما پاتشایی ٱتابٱکیا لۏر گٱپ و لۏر کوچک سی شومار نوفۊس و ٱنازٱ ڤلاتی ڤنو دیارکاری بینٱ. گلال دز یا سزار اؽ دۏ پاتشایی ناْ بٱر مؽکرد ڤ دۏ کوت لۏر کوچک د شٱمال ۉ لٛۏر گٱپ د هارگٱ.


د مؽنجایا سٱدٱ شٱش مالکنون سلسلٱ ٱتابٱکان لٛۏر گٱپ یا اومٱرا فٱزلۊیٱ د لٛۏرسوݩ گٱپ دۏرس بی. ڤنو لٛۏرؽیاؽؽ بیئن کاْ د مؽنجایا سٱدٱ ۵ مالکٱنون دئما د باٛن رٱتن آلاْ بۊیٱ ۉ فرٱ بیئاْن ناٱمنی د راساگٱیا اؽروݩ فٱت الدین فٱزلۊیٱ ڤارداْ یٱکؽ د پاتشایا بٱدر ڤا چارسٱد هونٱڤار د لٛۏر گٱپ رٱتن سی جٱبٱل السوماق شام ۉ اۊچاْ جاگٱ پاتشایی خوشوݩ نؽان، ڤٱلٛی دئما د چٱن گات د آخرؽیا پٱنج مالکنوݩ ئٱلٛی کوراْ فٱت الدین فٱزلۊیٱ رٱتن سی مؽافارقین. د دۉر ۉ گرد سال ۵۵۰ مالکنوݩ ٱبۊتاهر سلسلٱ ٱتابٱکؽا گٱپ ناْ رٱڤٱندیاری کرد. آخری ٱتابٱک لٛۏر گٱپ غؽاس الدین بن کاڤۊس بن پٱشٱنڳ بی کاْ د سال ۸۲۷ مٱلکنوݩ سٱرنگۊن بی ۉ ٱتابٱکؽا لٛۏر گٱپ دئما ساْ سٱدٱ « سیسٱد سال» ڤ دٱس سۏلتوݩ ٱبراهیم بن شاروخ تاٛمۊری د سٱرنگۊن بی ۉ سلسلٱ ٱتابٱکؽا لٛۏر گٱپ تٱموم بی پاتشایا ٱتابٱک لٛؽزگاؽؽ سی نیسمون شناسؽا ۉ دونسمٱنؽا بی، ۉ گات ڤنو پور د یٱشنایی ۉ رٱزئشت سی ڤلات لٛۏر گٱپ بی. اؽ ڤلات ﺩ ﺳﺎ پاتشایا رٱزئشت دۊس دۊس، مؽنجاڤٱن دار ۉ دگشت دونسمٱنی ۉ یٱشنایی بی. دونسمٱنیجایا ۉ ٱتابٱکؽا لٛۏرسوݩ ۉ ﻣﺎﻝ ٱمیر مؽنجاڤٱن پاتشایی ڤنو، چی دونسمٱنیجایا ایسنی پور رۉنٱق ۉ دۊنسمٱنیا فراٛ د اۊچاْ ﻳﺎ آمۊختارکاری داشتن یا آمۊختارکار بیئن.


د سال ۵۷۰ مالکٱون شۏجا الدین خورشید بن ٱبۊبٱکر بن مهٱمٱد بن خورشید تونس بۊٱ ڤ پاتشا لٛۏرؽیا کوچک ۉ سلسلٱ ٱتابٱکؽیا لٛۏر کوچکاْ ڤا پاتٱختی ٱتابٱکؽیا شٱئر خورموئٱ رٱڤٱندیاری بٱکٱ. رٱڤٱندیار سلسلٱ ٱتابٱکؽیا لٛۏر کوچک شۏجا الدین خورشید پؽا کاردۏرس ۉ دلدارؽ بیٱ ۉ خوش سفٱت بیٱ، ڤا زؽر دٱسؽاش ڤا دلنٱرمی رٱفتار داشتٱ. اؽ پاتشایا د تایفاٛ لٛۏر جٱنڳرۊیی یا چٱنڳری بینٱ ڤ مانؽ جاگٱ زئشت ڤنو د راساگٱ تٱرهان د هارگٱ کۊدٱشت بیٱ ۉ نوماْ مال کۊیی ناْ ڤ نوماْ چٱنڳری ناْ ڤ ری خوشو نؽان. اؽ پاتشایی د سال ۱۰۰۶ مالکٱنون ڤا تۊزسسن شاڤردی خو آخری ٱتابٱک لۏر د شائٱباس سٱفٱڤی ۉ کوشسنش ڤ دٱس شائٱباس د باٛن رٱت.


دما د باٛن رٱتن ٱتابٱکؽا لٛۏر گٱپ یاٛ قٱرن ۉ نسمؽ دماتر د ڤیرگار دونسمٱنیایی د پاتشایی بٱختیوری د دٱس نؽ تا یٱ کاْ د گات شا تٱماس یٱکم سٱفٱڤی گرد خاکاْ لٛۏربٱختوری اوفتا ڤ دٱس یٱکؽ ڤاْ نوم تاج میر راٛیس تاؽفٱ آسترکی کاْ د او گات هؽزدارترین کٱسون بٱختیوری بی دراۊم. ۉ هٱر سال پیلٛؽ ڤ نوم مالؽات سی شا تٱماس سٱفٱڤی کلٛ مؽکرد ۉ سی یٱ کاْ د سال ۹۷۴ مالکنوݩ د مالؽات نٱدٱ شا تٱماس د پاتشایی بٱختیوری نیائش ڤا لٛا ۉ کوشتش. دما د ڤٱ میرجونگیرخوݩ نومؽ ناْ ڤ شٱرتاْ داٛین ۱۰۰۰ تا قاتر کرد ڤ پاتشا بٱختیوری، د او موهود بی کاْ بٱختوری بی ڤ دۏ کوت چارلٛنڳ هٱفت لٛنڳ کاْ جیاڤازیا فراٛ دۏرس بی تا گات هۏساٛقولٛی خوناْ بٱختیوری اؽ جیاڤازیا ناْ داشت.


دما کوشسناْ شۉردی خوݩ آخری ٱتابٱک لٛۊر شاباس کچیزا شۉردی خوݩ، هۏساٛ خوݩ کور مٱنسۊر باٛگ سلاٛڤرزی ناْ ڤا داسون ڤالٛی ڤ پاتشایی راساگٱیا لٛۏراْ کوچک گلٛٱڤرچاْ کرد. سلسلٱ ڤالٛیا لٛۏرسوݩ پاتشایی ٱتابٱکؽا لٛۏر کوچک د راساگٱیا لٛۏرنشی بی. دماتر د گات هٱسٱن خوݩ هکۊمٱت ڤنوناْ کرد ڤ دۏ قٱل پیشکۏ لٛۏرسوݩ ۉ پوشتکۏ لٛۏرسوݩ ۉ هکۊمٱت ڤالٛیا تٱنیا بی ڤ پوشتکۏ؛ د گاتی کاْ ڤالٛیا د گرداْ راساگٱیا لٛۏرسوݩ (پیشکۏ)ۉ ایلٛام (پوشتکۏ) پاتشایی مؽکردن، پاتٱختشو خورموئٱ ۉ د قٱلٱ دوؤزٱ بورجی بی ۉ دماتر کاْ پاتشایشو تٱنیا ڤ پوشتکۏ بی پاتٱختشۏ بی ڤ ایلام (هۏساٛناوؤٱ یا دؽڤالا). ڤا خلاس بین سلسلٱ ڤالٛیا لٛۏرسوݩ ڤ دٱس رزا شا پٱلٱڤی د سال ۱۳۰۸ دۉلٱت نیمٱ خودسلادار لٛۏریا د راساگٱیا لٛۏرنشیݩ د باٛن رٱت.


د گاتی کاْ نادرشا ٱفشار بی ڤ پاتشا سی د باٛن بوردن هۊزؽیا راساگٱیی لٛۏرؽیا ناْ د راساگٱ یایی چی قوم، کاشوݩ، خوراسوݩ ۉ آستونؽایؽ هٱنی مال د بار کرد. دما کوشسن نادرشا، هومالؽیا پاتشایی هٱر کوم سیکؽ د اؽروناْ ڤ دٱس آوؤردن. سی تا د سٱردارؽیا هیزدار نادرشا، ئٱلٛی مٱردو خوݩ بٱختیوری، کریم خوݩ زٱند ۉ اْسمایلٛ خوݩ فاٛیلٛی د تیرٱ لٛٱک بین کاْ هٱر ساْ د گٱپؽا تٱشؽا لٛۏر بیئن سی ڤ دٱس گرتن پاتشایی رٱتن د کار سیاسی. دما یٱکاْ باٛن ئٱلٛی مٱردو خوݩ بٱختیوری، ٱبولفٱت خوݩ بٱختیوری ۉ کریم خوݩ زٱند دئبارٱ آخری پاتشایی اؽروݩ، کریم خوݩ رٱت سی کشڤٱرگوشایی خوش ڤٱلؽ دما ینٱ، سی خیانٱت ئٱلٛ مٱردو خوݩ بٱختیوری ڤ ٱبولفٱت خوݩ بٱختیوری ۉ یوغی گٱری ٱبولفٱت خوݩ سی کریم خوݩ بائس جیاڤازی ۉ جٱنڳ بی. دماتر د ڤنو کریم خوݩ زٱند ڤ کولٛ هومالوناْ خوش زئلٛ بی ۉ د سالاْ ۱۱۶۳ مالکنوݩ بی ڤ پاتشا اؽروݩ ۉ سلسلٱ زٱندیٱ ناْ ڤٱن ڤ رٱ د جالی پاتشا ڤ خوش مؽگوت ڤٱکیلٛو روایا. کریم خوݩ د هۊز بٱگلٱ زٱند بی کاْ د لٛون لٛۏر ۉ تیرٱ لٛٱک بین کاْ د نزیکیا مٱلایر ۉ راساگاٛ ڤ نوم پٱری یا کٱمازو زئشت مؽکردن. کریم خوݩ زٱند نهاتر د یٱ مٱلایر ناْ کرد ڤ پاتٱختش ۉ سی یٱ کاْ نمؽتونس ڤ گرد اؽروݩ سۏلتٱ داشتۊ پایتٱختشاْ کرد ڤ شیراز. قری د اؽلؽیا لٛۏر د دۉرش بینی پاتشایی کریم‌خوݩ هیاری کردن ۉ ڤاردش رٱتن ڤ شیراز ڤٱلٛی دما ڤروفداین سلسلٱ زٱندیٱ گلٛ آوردن ڤ لٛۏرسوݩ.
